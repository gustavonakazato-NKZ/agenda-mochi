    <script type="module">

        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";

        import { getAuth, signInAnonymously, signInWithCustomToken, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";

        import { getFirestore, doc, addDoc, setDoc, getDoc, getDocs, onSnapshot, collection, query, where, deleteDoc, writeBatch } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

       

        const apiKey = "";

        const appId = typeof __app_id !== 'undefined' ? __app_id : 'agenda-dep-mochi-v2';

        const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : {};

        const initialAuthToken = typeof __initial_auth_token !== 'undefined' ? __initial_auth_token : null;

 

        let db, auth, currentUserData = null;

        let appointments = [];

 

        const loginScreen = document.getElementById('login-screen');

        const mainApp = document.getElementById('main-app');

        const adminTab = document.getElementById('tab-admin');

        const registerTab = document.getElementById('tab-form');

       

        try {

            const app = initializeApp(firebaseConfig);

            db = getFirestore(app);

            auth = getAuth(app);

           

            const initAuth = async () => {

                if (initialAuthToken) await signInWithCustomToken(auth, initialAuthToken);

                else await signInAnonymously(auth);

            };

            initAuth();

        } catch (e) { console.error(e); }

 

        // --- SISTEMA DE LOGIN COM DESATIVAÇÃO DO ACESSO INICIAL ---

        window.handleLogin = async (e) => {

            e.preventDefault();

            const cpfInput = document.getElementById('login-cpf').value.replace(/\D/g, '');

            const passInput = document.getElementById('login-pass').value;

 

            if (!cpfInput || !passInput) return alert("Preencha todos os campos");

 

            try {

                const userDoc = await getDoc(doc(db, 'artifacts', appId, 'public', 'data', 'users', cpfInput));

               

                if (userDoc.exists() && userDoc.data().password === passInput) {

                    currentUserData = userDoc.data();

                    loginSuccess();

                } else {

                    if (cpfInput === '00000000000' && passInput === 'admin123') {

                        const usersSnap = await getDocs(collection(db, 'artifacts', appId, 'public', 'data', 'users'));

                       

                        if (usersSnap.empty) {

                            currentUserData = { cpf: '00000000000', name: 'Administrador Inicial', role: 'Administrador' };

                            loginSuccess();

                        } else {

                            alert("O acesso administrativo inicial foi desativado por segurança pois já existem usuários cadastrados.");

                        }

                    } else {

                        alert("CPF ou Senha incorretos.");

                    }

                }

            } catch (err) { alert("Erro ao conectar ao servidor."); }

        };

 

        const loginSuccess = () => {

            loginScreen.classList.add('hidden');

            mainApp.classList.remove('hidden');

            setupPermissions();

            loadAppointments();

            if(currentUserData.role === 'Administrador') loadUsers();

        };

 

        const setupPermissions = () => {

            const role = currentUserData.role;

            document.getElementById('user-display-name').textContent = `${currentUserData.name} (${role})`;

            adminTab.classList.toggle('hidden', role !== 'Administrador');

            registerTab.classList.toggle('hidden', role === 'Visitante');

            document.getElementById('import-btn-container').classList.toggle('hidden', !['Administrador', 'Principal'].includes(role));

            document.getElementById('export-btn-container').classList.toggle('hidden', role === 'Visitante');

            if (role === 'Visitante') switchView('list');

            else switchView('form');

        };

 

        // --- GERENCIAMENTO DE USUÁRIOS (Admin) ---

        window.saveUser = async (e) => {

            e.preventDefault();

            const cpf = document.getElementById('user-cpf').value.replace(/\D/g, '');

            const name = document.getElementById('user-name').value;

            const pass = document.getElementById('user-pass').value;

            const role = document.getElementById('user-role').value;

 

            try {

                await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'users', cpf), {

                    cpf, name, password: pass, role

                });

                alert("Usuário cadastrado!");

                document.getElementById('user-form').reset();

            } catch (e) { alert("Erro ao salvar usuário"); }

        };

 

        const loadUsers = () => {

            const q = query(collection(db, 'artifacts', appId, 'public', 'data', 'users'));

            onSnapshot(q, (snapshot) => {

                const list = document.getElementById('users-list');

                list.innerHTML = '';

                snapshot.forEach(doc => {

                    const u = doc.data();

                    const div = document.createElement('div');

                    div.className = "p-3 bg-gray-50 rounded-lg flex justify-between items-center text-sm mb-2 border";

                    div.innerHTML = `<span><b>${u.name}</b> (${u.role})<br><small>CPF: ${u.cpf}</small></span>

                                    <button onclick="window.deleteUser('${u.cpf}')" class="text-red-500 font-bold">Excluir</button>`;

                    list.appendChild(div);

                });

            }, (error) => console.error("Erro ao carregar utilizadores:", error));

        };

 

        window.deleteUser = async (cpf) => {

            if(confirm("Excluir este usuário?")) await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'users', cpf));

        };

 

        // --- COMPROMISSOS ---

        const loadAppointments = () => {

            const q = query(collection(db, 'artifacts', appId, 'public', 'data', 'appointments'));

            onSnapshot(q, (snapshot) => {

                appointments = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));

                appointments.sort((a, b) => a.date.localeCompare(b.date) || a.time.localeCompare(b.time));

                renderAppointments();

            }, (error) => console.error("Erro ao carregar compromissos:", error));

        };

 

        const renderAppointments = () => {

            const list = document.getElementById('appointments-list');

            list.innerHTML = '';

            let currentDay = '';

            const filterValue = document.getElementById('list-filter').value;

            const today = new Date().toISOString().split('T')[0];

 

            let filtered = appointments.filter(app => {

                if (currentUserData.role === 'Visitante') return app.authorized === true && app.date >= today;

                if (filterValue === 'Ambos') return true;

                return (app.activity_type || 'Mandato') === filterValue;

            });

 

            filtered.forEach(app => {

                if (app.date !== currentDay) {

                    const h = document.createElement('h2');

                    h.className = 'text-lg font-bold mt-6 mb-2 border-b border-indigo-200 text-indigo-700';

                    h.textContent = formatDate(app.date);

                    list.appendChild(h);

                    currentDay = app.date;

                }

 

                const isHap = app.status === 'Acontecendo';

                const isAuth = app.authorized === true;

                const canAuth = ['Administrador', 'Principal'].includes(currentUserData.role);

 

                const card = document.createElement('div');

                card.className = `card p-4 mb-3 rounded-xl transition ${isAuth ? (isHap ? 'bg-red-50 ring-1 ring-red-300' : 'bg-white') : 'unauthorized-badge'}`;

               

                let timeColor = isHap ? 'text-red-700' : ((app.activity_type === 'Campanha') ? 'text-green-900 font-bold' : 'text-indigo-600');

 

                card.innerHTML = `

                    <div class="flex items-start">

                        <div class="text-xl font-black w-16 flex-shrink-0 ${timeColor}">${app.time}</div>

                        <div class="flex-grow">

                            <p class="font-bold text-gray-800">${app.details}</p>

                            <div class="grid grid-cols-1 md:grid-cols-2 gap-x-4 mt-2 text-xs text-gray-500">

                                <div><span class="font-semibold">Local:</span> ${app.location || '-'}</div>

                                <div><span class="font-semibold">Atividade:</span> ${app.activity_type || 'Mandato'}</div>

                                <div><span class="font-semibold">Imprensa:</span> ${app.press_coverage === 'Acompanhar' ? '<u>Acompanhar</u>' : 'Não Acompanhar'}</div>

                                ${!isAuth ? '<div class="text-yellow-700 font-bold">AGUARDANDO AUTORIZAÇÃO</div>' : ''}

                            </div>

                        </div>

                        <div class="flex space-x-1 ml-2">

                            ${canAuth ? `<button onclick="window.toggleAuth('${app.id}', ${!isAuth})" class="p-1 ${isAuth ? 'text-gray-400' : 'text-green-600 font-bold'} rounded" title="Autorizar">${isAuth ? '🔓' : '✅'}</button>` : ''}

                            ${currentUserData.role !== 'Visitante' ? `

                                <button onclick="window.duplicateApp('${app.id}')" class="p-1 text-purple-600 rounded">❐</button>

                                <button onclick="window.editApp('${app.id}')" class="p-1 text-blue-600 rounded">✎</button>

                                <button onclick="window.deleteApp('${app.id}')" class="p-1 text-red-600 rounded">✕</button>

                            ` : ''}

                        </div>

                    </div>

                `;

                list.appendChild(card);

            });

        };

 

        window.saveAppointment = async (e) => {

            e.preventDefault();

            const id = document.getElementById('appointment-id').value;

            const isPrivileged = ['Administrador', 'Principal'].includes(currentUserData.role);

           

            const data = {

                date: document.getElementById('appointment-date').value,

                time: document.getElementById('appointment-time').value,

                details: document.getElementById('appointment-details').value.trim(),

                press_coverage: document.getElementById('press-coverage').value,

                contact: document.getElementById('appointment-contact').value.trim(),

                activity_type: document.getElementById('activity-type').value,

                location: document.getElementById('appointment-location').value.trim(),

                status: document.getElementById('appointment-status').checked ? 'Acontecendo' : '',

                authorized: isPrivileged ? true : false

            };

 

            try {

                const colRef = collection(db, 'artifacts', appId, 'public', 'data', 'appointments');

                if (id) await setDoc(doc(colRef, id), data, { merge: true });

                else await addDoc(colRef, data);

               

                alert("Compromisso registrado!");

                document.getElementById('appointment-form').reset();

                document.getElementById('appointment-id').value = '';

                document.getElementById('appointment-date').value = new Date().toISOString().split('T')[0];

            } catch (e) { alert("Erro ao salvar"); }

        };

 

        window.toggleAuth = async (id, status) => {

            await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'appointments', id), { authorized: status }, { merge: true });

        };

 

        window.exportPDFTable = () => {

            const start = document.getElementById('exp-start').value;

            const end = document.getElementById('exp-end').value;

            const typeFilter = document.getElementById('exp-filter').value;

           

            let filtered = appointments.filter(a => a.authorized === true && a.date >= start && a.date <= end);

            if (typeFilter !== 'Ambos') filtered = filtered.filter(a => (a.activity_type || 'Mandato') === typeFilter);

 

            if (!filtered.length) return alert("Nenhum compromisso autorizado encontrado para o período.");

 

            let printHTML = `<div class="print-container"><h1 style="text-align:center">Agenda - Deputado Junior Mochi</h1>`;

            const grouped = {};

            filtered.forEach(app => {

                if (!grouped[app.date]) grouped[app.date] = [];

                grouped[app.date].push(app);

            });

 

            Object.keys(grouped).sort().forEach(date => {

                printHTML += `

                    <div class="date-header">DATA: ${formatDate(date)}</div>

                    <table>

                        <thead>

                            <tr><th>Horário - Atividade</th><th>Compromisso</th><th>Imprensa</th><th>Contato</th><th>Local</th></tr>

                        </thead>

                        <tbody>

                `;

                grouped[date].forEach(app => {

                    const rowStyle = app.status === 'Acontecendo' ? 'color:red; font-weight:bold' : '';

                    printHTML += `

                        <tr style="${rowStyle}">

                            <td>${app.time} - ${app.activity_type || 'Mandato'}</td>

                            <td>${app.details}</td>

                            <td>${app.press_coverage}</td>

                            <td>${app.contact || '-'}</td>

                            <td>${app.location || '-'}</td>

                        </tr>`;

                });

                printHTML += `</tbody></table>`;

            });

           

            const win = window.open('', '_blank');

            win.document.write(`<html><head><style>body{font-family:sans-serif;padding:20px} table{width:100%;border-collapse:collapse;margin-bottom:20px} th,td{border:1px solid #000;padding:6px;font-size:9pt} .date-header{background:#eee;padding:4px;font-weight:bold;border:1px solid #000}</style></head><body>${printHTML}</body></html>`);

            win.document.close();

            win.print();

        };

 

        const switchView = (viewName) => {

            document.getElementById('form-view').classList.toggle('hidden', viewName !== 'form');

            document.getElementById('list-view').classList.toggle('hidden', viewName !== 'list');

            document.getElementById('admin-view').classList.toggle('hidden', viewName !== 'admin');

           

            const tabs = { 'form': 'tab-form', 'list': 'tab-list', 'admin': 'tab-admin' };

            Object.keys(tabs).forEach(k => {

                const el = document.getElementById(tabs[k]);

                if (!el) return;

                if (k === viewName) { el.classList.add('bg-indigo-600', 'text-white'); el.classList.remove('bg-gray-200'); }

                else { el.classList.remove('bg-indigo-600', 'text-white'); el.classList.add('bg-gray-200'); }

            });

        };

 

        const formatDate = (d) => d.split('-').reverse().join('/');

 

        window.onload = () => {

            document.getElementById('appointment-date').value = new Date().toISOString().split('T')[0];

            document.getElementById('tab-form').onclick = () => switchView('form');

            document.getElementById('tab-list').onclick = () => switchView('list');

            document.getElementById('tab-admin').onclick = () => switchView('admin');

            document.getElementById('list-filter').onchange = renderAppointments;

        };

 

        window.editApp = (id) => {

            const app = appointments.find(a => a.id === id);

            if(!app) return;

            switchView('form');

            document.getElementById('appointment-id').value = app.id;

            document.getElementById('appointment-date').value = app.date;

            document.getElementById('appointment-time').value = app.time;

            document.getElementById('appointment-details').value = app.details;

            document.getElementById('appointment-contact').value = app.contact || '';

            document.getElementById('activity-type').value = app.activity_type || 'Mandato';

            document.getElementById('press-coverage').value = app.press_coverage;

            document.getElementById('appointment-location').value = app.location || '';

            document.getElementById('appointment-status').checked = app.status === 'Acontecendo';

            document.getElementById('form-title').textContent = "Editar Compromisso";

        };

 

        window.deleteApp = async (id) => {

            if(confirm("Excluir?")) await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'appointments', id));

        };

 

        window.duplicateApp = (id) => {

            window.editApp(id);

            document.getElementById('appointment-id').value = '';

            document.getElementById('form-title').textContent = "Duplicar Compromisso";

        };

    </script>
